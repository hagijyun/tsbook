predict.dlm <- function(object, interval = "confidence", level = 0.95, states = c(1)){
  # 
  band <- 1 - level; conf_l <- 0 + band/2; conf_u <- 1 - band/2

  # 
  obj_name <- names(object)

  # 
  if ("U.C" %in% obj_name){
    # 
    m <- dropFirst(object$m)                                                          # 
    m_ts <- m                                                                         # ts
    if (is.matrix(m) == FALSE){ m <- matrix(m, ncol = 1) }                            # 11
    sum_m <- rowSums(m[, states, drop = FALSE])                                       # 

    # 
    C <- dlmSvd2var(object$U.C, object$D.C); C <- C[-1]                               # 
    sum_sdev <- sapply(C, function(cov_mat){ sqrt(sum(cov_mat[states, states])) })    # 

    # 
    res <- cbind(sum_m, sum_m + qnorm(conf_l, sd = sum_sdev), sum_m + qnorm(conf_u, sd = sum_sdev))
    colnames(res) <- c("mean", "lower", "upper")                                      # 
    if (is.ts(m_ts) == TRUE){ res <- ts(res); tsp(res) <- tsp(m_ts) }                 # ts

    return(res)
  # 
  }else if ("U.S" %in% obj_name){
    # 
    s <- dropFirst(object$s)                                                          # 
    s_ts <- s                                                                         # ts
    if (is.matrix(s) == FALSE){ s <- matrix(s, ncol = 1) }                            # 11
    sum_s <- rowSums(s[, states, drop = FALSE])                                       # 

    # 
    S <- dlmSvd2var(object$U.S, object$D.S); S <- S[-1]                               # 
    sum_sdev <- sapply(S, function(cov_mat){ sqrt(sum(cov_mat[states, states])) })    # 

    # 
    res <- cbind(sum_s, sum_s + qnorm(conf_l, sd = sum_sdev), sum_s + qnorm(conf_u, sd = sum_sdev))
    colnames(res) <- c("mean", "lower", "upper")                                      # 
    if (is.ts(s_ts) == TRUE){ res <- ts(res); tsp(res) <- tsp(s_ts) }                 # ts

    return(res)
  # 
  }else if ("R" %in% obj_name){
    # 
    a <- object$a                                                                     # 
    a_ts <- a                                                                         # ts
    if (is.matrix(a) == FALSE){ a <- matrix(a, ncol = 1) }                            # 11
    sum_a <- rowSums(a[, states, drop = FALSE])                                       # 

    # 
    R <- object$R                                                                     # 
    sum_sdev <- sapply(R, function(cov_mat){ sqrt(sum(cov_mat[states, states])) })    # 

    # 
    res <- cbind(sum_a, sum_a + qnorm(conf_l, sd = sum_sdev), sum_a + qnorm(conf_u, sd = sum_sdev))
    colnames(res) <- c("mean", "lower", "upper")                                      # 
    if (is.ts(a_ts) == TRUE){ res <- ts(res); tsp(res) <- tsp(a_ts) }                 # ts

    return(res)
  # 
  }else{
    warning("object")
    return(NA)
  }
}
